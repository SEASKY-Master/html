1. SEASKY串口通信协议¶

注解

串口通信协议，包含帧头CRC8校验以及数据的整包校验，可以实现不定长收发数据（例如RoboMaster中可用于和视觉双方通信，以及其他自研模块）

1.1. 串口配置¶

通信方式是串口，配置为波特率 115200， 8 位数据位， 1 位停止位， 无 硬件流控， 无 校验位。

提示

使用说明，首先将接收数据存储于 rx_buf[],每接收一个数据 *rx_pos 的值会加一，以便于存储下一个数据，原始数据缓冲的存储由中断方式完成，在linux或win上可能有所区别，调用此函数，函数中包含crc校验，数据解读等相关处理，最后会更新 flags_register 的值即为收到的16位寄存器的值， rx_data[] 即为收到的 float 数据；

 10/*获取CRC8校验码*/
 11uint8_t Get_CRC8_Check(uint8_t *pchMessage,uint16_t dwLength)
 12{
 13    return crc_8(pchMessage,dwLength);
 14}
 15/*检验CRC8数据段*/
 16uint8_t CRC8_Check_Sum(uint8_t *pchMessage,uint16_t dwLength)
 17{
 18    uint8_t ucExpected = 0;
 19    if ((pchMessage == 0) || (dwLength <= 2)) return 0;
 20    ucExpected = Get_CRC8_Check(pchMessage, dwLength-1);
 21    return ( ucExpected == pchMessage[dwLength-1] );
 22}
 23/*获取CRC16校验码*/
 24uint16_t Get_CRC16_Check(uint8_t *pchMessage,uint32_t dwLength)
 25{
 26    return crc_16(pchMessage,dwLength);
 27}
 28/*检验CRC16数据段*/
 29uint16_t CRC16_Check_Sum(uint8_t *pchMessage, uint32_t dwLength)
 30{
 31    uint16_t  wExpected = 0;
 32    if ((pchMessage == 0) || (dwLength <= 2))
 33        {
 34            return 0;
 35        }
 36    wExpected = Get_CRC16_Check ( pchMessage, dwLength - 2);
 37    return (((wExpected & 0xff) == pchMessage[dwLength - 2] )&& (((wExpected >> 8) & 0xff) == pchMessage[dwLength - 1]));
 38}
 39/*检验数据帧头*/
 40uint8_t protocol_heade_Check(
 41    protocol *pro,
 42    uint8_t  *rx_buf,
 43    uint16_t *rx_pos)
 44{
 45    if(rx_buf[0] == PROTOCOL_CMD_ID)
 46        {
 47            pro->header.sof = rx_buf[0];
 48            if(CRC8_Check_Sum(&rx_buf[0],4))
 49                {
 50                    pro->header.data_length = (rx_buf[2]<<8) | rx_buf[1];
 51                    pro->header.crc_check = rx_buf[3];
 52                    pro->cmd_id = (rx_buf[5]<<8) | rx_buf[4];
 53                    return  1;
 54                }
 55        }
 56    else
 57        {
 58            *rx_pos = 0;
 59        }
 60    return 0;
 61}

 63float float_protocol(uint8_t *dat_t)
 64{
 65    uint8_t f_data[4];
 66    f_data[0] = *(dat_t+0);
 67    f_data[1] = *(dat_t+1);
 68    f_data[2] = *(dat_t+2);
 69    f_data[3] = *(dat_t+3);
 70    return *(float*)f_data;
 71}
 72/*
 73  此函数根据待发送的数据更新数据帧格式以及内容，实现数据的打包操作
 74  后续调用通信接口的发送函数发送tx_buf中的对应数据
 75*/
 76void get_protocol_send_data
 77(uint16_t send_id,              //信号id
 78uint16_t flags_register, //16位寄存器
 79float    *tx_data,              //待发送的float数据
 80uint8_t  float_length,   //float的数据长度
 81uint8_t  *tx_buf,               //待发送的数据帧
 82uint16_t *tx_buf_len)   //待发送的数据帧长度
 83{
 84    uint16_t crc16;
 85    uint16_t data_len;
 86
 87    data_len = float_length*4+2;
 88    /*帧头部分*/
 89    tx_buf[0] = PROTOCOL_CMD_ID;
 90    tx_buf[1] = data_len & 0xff;                    //低位在前
 91    tx_buf[2] = (data_len >> 8) & 0xff;       //低位在前
 92    tx_buf[3] = Get_CRC8_Check(&tx_buf[0],3); //获取CRC8校验位
 93
 94    /*数据的信号id*/
 95    tx_buf[4] = send_id & 0xff;
 96    tx_buf[5] = (send_id>>8) & 0xff;
 97
 98    /*建立16位寄存器*/
 99    tx_buf[6] = flags_register & 0xff;
100    tx_buf[7] = (flags_register>>8) & 0xff;
101
102    /*float数据段*/
103    for(int i=0; i<4*float_length; i++)
104        {
105            tx_buf[i+8]  = ((uint8_t*)(&tx_data[i/4]))[i%4];
106        }
107
108    /*整包校验*/
109    crc16 = Get_CRC16_Check(&tx_buf[0],data_len+6);
110    tx_buf[data_len+6] = crc16 & 0xff;
111    tx_buf[data_len+7] = (crc16 >>8) & 0xff;
112
113    *tx_buf_len = data_len+8;
114}
115/*
116  此函数用于处理接收数据，
117  返回数据内容的id
118*/
119
120uint16_t get_protocol_info
121(uint8_t  *rx_buf,                    //接收到的原始数据
122uint16_t *rx_pos,                     //原始数据指针
123uint16_t *flags_register,     //接收数据的16位寄存器地址
124float    *rx_data)                    //接收的float数据存储地址
125{
126  static protocol pro;
127  static uint16_t date_length;
128    if(protocol_heade_Check(&pro,rx_buf,rx_pos))
129        {
130            date_length = OFFSET_BYTE + pro.header.data_length;
131            while(CRC16_Check_Sum(&rx_buf[0],date_length))
132                {
133                    *flags_register = (rx_buf[7]<<8) | rx_buf[6];
134                    for(int i=0; i<(pro.header.data_length-2)/4; i++)
135                        {
136                            rx_data[i] = float_protocol(&rx_buf[8+4*i]);
137                        }
138                    for(int i=0; i<date_length; i++)
139                        {
140                            rx_buf[i] = 0;
141                        }
142                    *rx_pos = 0;
143                    return pro.cmd_id;
144                }
145        }
146}

154            *rx_buf_pos = 0;
